# ============================================================================
# JAKE RL → RATIO LATTICE INDEX
# ============================================================================
# 
# PLAN:
# 
# The JakeRL.txt file is a psychological/cognitive profile of Jake organized 
# into Q&A sections. The Ratio Lattice is a CRM sales intelligence system 
# with 40 schema files.
#
# The question: Can we build an "index" that maps the cognitive/philosophical 
# traits identified in JakeRL.txt onto the architectural features of the 
# Ratio Lattice — showing how the person's mind is literally encoded into 
# the system's design?
#
# Approach:
# 1. Extract the key cognitive/philosophical traits from JakeRL.txt
# 2. Extract the key architectural features from the Ratio Lattice
# 3. Build a mapping: for each trait, which lattice features embody it?
# 4. Score the strength of each mapping
# 5. Create a structured index + final visualization
#
# This is an interpretive/structural analysis, not a statistical test.

print("Step 1: Parse JakeRL.txt into structured traits")
print("="*60)

# Read the full JakeRL file
with open("JakeRL.txt", "r") as f:
    jake_text = f.read()

# Split into sections by the numbered question pattern
import re

# Each section starts with a chapter (separated by numbered lists starting from 1)
# Let's find all Q&A pairs
qa_pairs = []
lines = jake_text.split('\n')

current_q = None
current_a = None

for line in lines:
    line = line.strip()
    if not line:
        if current_q and current_a:
            qa_pairs.append((current_q, current_a))
            current_q = None
            current_a = None
        continue
    
    # Check if this is a question line (starts with number followed by period)
    q_match = re.match(r'^\d+\.\s+(.+)', line)
    if q_match:
        if current_q and current_a:
            qa_pairs.append((current_q, current_a))
        current_q = q_match.group(1)
        current_a = None
    elif current_q and current_a is None:
        current_a = line
    elif current_q and current_a:
        current_a += " " + line

if current_q and current_a:
    qa_pairs.append((current_q, current_a))

print(f"Extracted {len(qa_pairs)} Q&A pairs from JakeRL.txt")
print(f"\nSample entries:")
for i in [0, 5, 20, 50, 100, 150]:
    if i < len(qa_pairs):
        print(f"\n  [{i}] Q: {qa_pairs[i][0][:80]}...")
        print(f"       A: {qa_pairs[i][1][:80]}...")

# ============================================================================
# Step 3: Parse the Ratio Lattice into its architectural features
# ============================================================================

import json

with open("RATIO_LATTICE_FULL_BUNDLE.txt", "r") as f:
    lattice_text = f.read()

# Extract all schema names and their core architectural features
# We'll parse each FILE section

file_sections = re.split(r'═{70,}\nFILE: (.+?)\n═{70,}', lattice_text)

lattice_schemas = {}
for i in range(1, len(file_sections), 2):
    filename = file_sections[i].strip()
    content = file_sections[i+1].strip() if i+1 < len(file_sections) else ""
    
    # Try to extract schema_name
    schema_match = re.search(r'"schema_name":\s*"([^"]+)"', content)
    schema_name = schema_match.group(1) if schema_match else filename
    
    # Extract description
    desc_match = re.search(r'"description":\s*"([^"]+)"', content)
    description = desc_match.group(1) if desc_match else ""
    
    # Extract field names
    fields = re.findall(r'"(\w+)":\s*"(?:string|number|boolean|datetime|jsonb)', content)
    
    # Extract sales_scoring fields
    scoring_match = re.search(r'"sales_scoring":\s*\{([^}]+)\}', content)
    scoring_fields = re.findall(r'"(\w+)":', scoring_match.group(1)) if scoring_match else []
    
    # Extract derived flags
    flags_match = re.search(r'"derived_flags":\s*\{([^}]+)\}', content)
    derived_flags = re.findall(r'"(\w+)":', flags_match.group(1)) if flags_match else []
    
    # Extract relationships
    rels_match = re.search(r'"relationships":\s*\{([^}]+)\}', content)
    relationships = re.findall(r'"(\w+)":', rels_match.group(1)) if rels_match else []
    
    lattice_schemas[schema_name] = {
        "filename": filename,
        "description": description,
        "fields": fields,
        "scoring_fields": scoring_fields,
        "derived_flags": derived_flags,
        "relationships": relationships,
        "raw_content": content[:2000]  # Keep first 2000 chars for reference
    }

print(f"RATIO LATTICE SCHEMA INVENTORY")
print("="*70)
print(f"Total schemas parsed: {len(lattice_schemas)}")
print(f"\n{'Schema':<35} {'Fields':>7} {'Scores':>7} {'Flags':>7} {'Rels':>5}")
print("-"*70)

for name, info in sorted(lattice_schemas.items()):
    print(f"  {name:<33} {len(info['fields']):>5} {len(info['scoring_fields']):>7} "
          f"{len(info['derived_flags']):>7} {len(info['relationships']):>5}")
